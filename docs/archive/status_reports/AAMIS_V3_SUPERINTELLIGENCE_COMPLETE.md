## 🔧 USAGE

### Basic Usage
```python
from trading_bot.aamis_v3.superintelligence import SuperintelligenceOrchestrator
import asyncio

async def main():
    # Initialize
    si = SuperintelligenceOrchestrator()
    
    # Market data
    market_data = {
        'macro': {'gdp_growth': 2.5, 'cpi': 3.0},
        'microstructure': {'price': 4500, 'vpoc': 4490},
        'sentiment': {'vix': 16},
        'current_state': {'regime': 'trending'}
    }
    
    # Analyze with superintelligence
    report = await si.analyze_with_superintelligence(market_data)
    
    print(f"Decision: {report.decision.action}")
    print(f"Conviction: {report.decision.conviction:.1f}%")
    print(f"Regime: {report.decision.detected_regime}")
    print(f"Strategy: {report.decision.active_strategy}")
    print(f"Intelligence: {report.decision.intelligence_score:.3f}")
    
    # Learn from outcome
    trade_outcome = {
        'action': 'BUY',
        'pnl': 150.0,
        'outcome': 'WIN',
        'what_worked': ['Trend following', 'Volume confirmation']
    }
    
    await si.learn_from_outcome(trade_outcome)
    
    # Daily evolution
    improvements = await si.daily_evolution_cycle()
    print(f"Intelligence improved by: {improvements['intelligence_improvement']:.4f}")

asyncio.run(main())
```
